# FvTh

<Canvas>
<Story name='Common'>{{
  components: { FvTh },
  data() {
    return {
      sortKey: 'test',
      isSorted: false,
      isSortedDesc: false
    }
  },
  props: {
    knobText: {
      default: text('Text', 'header', 'Props')
    },
    knobSortable: {
      default: boolean('Sortable', true, 'Props')
    }
  },
  methods: {
    thSort(v) {
      if (this.isSorted && this.isSortedDesc) {
        this.isSorted = false
        this.isSortedDesc = false
      } else if (this.isSorted) {
        this.isSortedDesc = true
      } else {
        this.isSorted = true
        this.isSortedDesc = false
      }
    }
  },
  template: `<FvTh
    :text="knobText"
    :sortKey="sortKey"
    :sortable="knobSortable"
    :isSorted="isSorted"
    :isSortedDesc="isSortedDesc"
    @th:sort="thSort"
  >
  </FvTh>`
}}</Story>
</Canvas>
